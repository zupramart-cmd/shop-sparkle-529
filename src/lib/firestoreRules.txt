rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Users
    // - Authenticated users can read user docs (needed for referral code lookup).
    // - A user can create/update their own doc; admins can update any.
    // - Loyalty points and referral rewards are credited ONLY by an admin, when an
    //   order's status is set to "delivered" (see updateOrderStatus / rewardReferrerIfPending
    //   in src/hooks/useFirestoreData.ts). No other user may modify someone else's
    //   loyaltyPoints, referredBy, referrerUserId, or referralRewardPending fields.
    // - Admin can create any user doc (needed for backup import/restore).
    match /users/{userId} {
      allow read: if request.auth != null;
      allow create: if request.auth != null && (request.auth.uid == userId || isAdmin());
      allow update: if request.auth != null && (
        request.auth.uid == userId
        || isAdmin()
      );
      allow delete: if isAdmin();
    }

    // Products
    match /products/{productId} {
      allow read: if true;
      allow write: if isAdmin();
    }

    // Categories
    match /categories/{categoryId} {
      allow read: if true;
      allow write: if isAdmin();
    }

    // Banners
    match /banners/{bannerId} {
      allow read: if true;
      allow write: if isAdmin();
    }

    // Coupons
    // - Everyone can read (to validate codes).
    // - Admin full write (including backup import/restore).
    // - A signed-in user can delete THEIR OWN coupon (consumed after checkout).
    // - A signed-in user can create their own loyalty coupon (userId == auth.uid).
    match /coupons/{couponId} {
      allow read: if true;
      allow create: if isAdmin() || (request.auth != null && request.resource.data.userId == request.auth.uid);
      allow delete: if isAdmin() || (request.auth != null && resource.data.userId == request.auth.uid);
      allow update: if isAdmin();
    }

    // Carts
    match /carts/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }

    // Orders
    // - A user can read their own orders; admins can read all.
    // - Any authenticated user can create an order (during checkout).
    // - Admin can update/delete orders (status management + backup import/restore).
    match /orders/{orderId} {
      allow read: if request.auth != null && (resource.data.userId == request.auth.uid || isAdmin());
      allow create: if request.auth != null;
      allow update, delete: if isAdmin();
    }

    // Settings
    match /settings/{settingId} {
      allow read: if true;
      allow write: if isAdmin();
    }

    // ── Backup / Restore ──────────────────────────────────────────────────────
    // The exportAllData() function only reads collections already covered above.
    // The importAllData() function calls setDoc(..., { merge: true }) on those
    // same collections, which maps to create-or-update.  All of those paths
    // already grant admin full write, EXCEPT:
    //   • users  → admin can now `create` any userId doc  (added above)
    //   • coupons → admin can now `create` any coupon doc (added above)
    // No extra collection rules are needed; the changes above are sufficient.
    // ─────────────────────────────────────────────────────────────────────────

    function isAdmin() {
      return request.auth != null &&
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin';
    }
  }
}
